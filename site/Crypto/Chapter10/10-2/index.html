# 把字符串 '010101' 转换为整数列表 [0,1,0,1,0,1]
def string2bits(s):
    return [int(b) for b in s]

# 这是一个线性反馈函数（非模 2）
def lfsr(state, mask):
    assert(len(state) == 128)
    assert(len(mask)  == 128)

    output = 0
    for i in range(128):
        # 两个向量点积
        output = output + (state[i] * mask[i])

    return output
